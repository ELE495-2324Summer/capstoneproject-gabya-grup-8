#include <Wire.h>
#include "Adafruit_TCS34725.h"

#define OUTPUT_PIN 7

// Create the TCS34725 object
Adafruit_TCS34725 tcs = Adafruit_TCS34725(TCS34725_INTEGRATIONTIME_50MS, TCS34725_GAIN_1X);

enum State {
  INITIAL,
  RED_DETECTED
};

State currentState = INITIAL;

void setup() {
  //Serial.begin(9600);
  
  // Set the output pin mode
  pinMode(OUTPUT_PIN, OUTPUT);
  digitalWrite(OUTPUT_PIN, LOW);
  
  // Initialize the sensor
  /*
  if (tcs.begin()) {
    Serial.println("Found sensor");
  } else {
    Serial.println("No TCS34725 found ... check your connections");
    while (1); // Halt!
  }

*/
}
void loop() {
  uint16_t r, g, b, c;
  
  // Get the RGB values from the sensor
  tcs.getRawData(&r, &g, &b, &c);
  
  // Print the values to the Serial Monitor
  //Serial.print("R: "); Serial.print(r); Serial.print(" ");
  //Serial.print("G: "); Serial.print(g); Serial.print(" ");
  //Serial.print("B: "); Serial.print(b); Serial.print(" ");
  //Serial.print("C: "); Serial.print(c); Serial.println(" ");
  
  if (currentState == INITIAL) {
    // Threshold for detecting a red line
    if (r > 55 && r < 80 && g < 50 && b < 50) {
      Serial.println("Red line detected! Transitioning to RED_DETECTED state.");
      digitalWrite(OUTPUT_PIN, HIGH);
      currentState = RED_DETECTED;
    }
  } else if (currentState == RED_DETECTED) {
    // Threshold for detecting a white color (all values high)
    if (r < 55 || r > 80) {
      Serial.println("White detected! Transitioning to INITIAL state.");
      digitalWrite(OUTPUT_PIN, LOW);
      currentState = INITIAL;
    }
  }
  
  // Delay before the next reading
  delay(10);
}
